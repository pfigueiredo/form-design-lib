Build Dynamic Forms from JSON Configuration

A powerful React component library for creating dynamic, API-integrated forms with built-in validation, conditional fields, and extensible field types. Perfect for CMS, admin panels, and enterprise applications.

Examples

Basic Form

const config = {
  formName: 'Contact Form',
  fields: [
    {
      fieldName: 'Name',
      fieldType: 'Text',
      binding: 'object.name',
      required: true,
    },
    {
      fieldName: 'Email',
      fieldType: 'Email',
      binding: 'object.email',
      required: true,
    },
  ],
}

API Integration

sources: [
  {
    sourceName: 'stateList',
    sourceType: 'API',
    endpoint: '/api/states?country={code}',
    parameters: {
      code: 'object.country',
    },
  },
]

Custom Field

import { defaultFieldRegistry } from 'form-design-lib'

defaultFieldRegistry.register(
  'Color',
  CustomColorField
)

// Use in config
{ fieldType: 'Color', binding: 'object.color' }

Getting Started

1

Install

npm install form-design-lib
2

Import

import { DynamicForm, FormConfig } from 'form-design-lib'
import 'form-design-lib/style.css'
3

Configure

const config: FormConfig = {
  formName: 'My Form',
  gridSize: 12,
  sources: [...],
  fields: [...],
}
4

Use

<DynamicForm
  config={config}
  apiProvider={apiProvider}
  onSubmit={(data) => console.log(data)}
/>
